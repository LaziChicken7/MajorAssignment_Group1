    %% =======================
    %% ENUMS
    %% =======================
    class Role {
        <<enumeration>>
        SELLER
        BIDDER
        ADMIN
    }

    class AuctionStatus {
        <<enumeration>>
        OPEN
        RUNNING
        FINISHED
        PAID
        CANCELLED
    }

    class ItemType {
        <<enumeration>>
        VEHICLE
        ELECTRONICS
        ART
    }

    class TransactionStatus {
        <<enumeration>>
        SUCCESS
        FAILED
        CANCELLED
    }

    class NotificationType {
        <<enumeration>>
        PAYMENT_VERIFICATION
        AUCTION_SUCCESS
        AUCTION_FAILED
        ITEM_CANCELLED_BY_ADMIN
        FRIEND_REQUEST
    }

    %% =======================
    %% BASE ENTITY
    %% =======================
    class BaseEntity {
        <<abstract>>
        #String id
        +getId() String
        +setId(String)
    }

    %% =======================
    %% USER PACKAGE
    %% =======================
    class User {
        #String userCode
        #String userName
        #String fullName
        #String email
        #String password
        #String numberPhone
        #String citizenId
        #Role role
        #String avatarUrl
        #boolean isBanned
        +encode(password) String$
        +decode(encodedPassword) String$
    }

    class Admin {
        -String department
    }

    class Bidder {
        #String address
        #String bankAccountNumber
        #BigDecimal moneyOnWallet
        #BigDecimal moneyinFrozen
        #List~Item~ successBidItem
        #List~Item~ failedBidItem
        +Deposit(money)
        +Withdraw(money)
        +addSuccessBidItem(item)
        +addFailedBidItem(item)
    }

    class Seller {
        -double rating
    }

    class SellerReview {
        -int star
        -String comment
        -LocalDateTime createdAt
    }

    class UserFactory {
        +createUser(role, ...) User$
    }

    BaseEntity <|-- User
    User <|-- Admin
    User <|-- Bidder
    Bidder <|-- Seller
    BaseEntity <|-- SellerReview
    User "1" *-- "1" Role
    UserFactory ..> User : creates

    SellerReview "1" --> "1" Seller : seller
    SellerReview "1" --> "1" User : reviewer

    %% =======================
    %% ITEM PACKAGE
    %% =======================
    class Item {
        #String name
        #String description
        #LocalDateTime startTime
        #BigDecimal startPrice
        #BigDecimal endPrice
        #ItemType itemType
    }

    class Art {
        -String nameAuthor
        -int creationYear
    }

    class Electronics {
        -String brand
        -int warrantyMonths
    }

    class Vehicle {
        -String engineType
        -int mileage
    }

    class ItemImageUrl {
        -String imageUrl
    }

    class ItemFactory {
        +createItem(type, ...) Item$
    }

    BaseEntity <|-- Item
    Item <|-- Art
    Item <|-- Electronics
    Item <|-- Vehicle
    Item "*" --> "1" Seller : owned by
    ItemFactory ..> Item : creates
    ItemFactory ..> ItemType : uses
    Item "1" o-- "*" ItemImageUrl : imageUrls

    %% =======================
    %% AUCTION PACKAGE
    %% =======================
    class Auction {
        -LocalDateTime startTime
        -LocalDateTime endTime
        -BigDecimal highestBid
        -BigDecimal stepPrice
        -AuctionStatus status
        -TransactionStatus transactionStatus
        -BigDecimal myHighestBid <<transient>>
        +placeBid(bidder, amount, time) boolean
        +startAuction()
        +closeAuction() boolean
        +acceptPayment()
        +cancelAuction()
    }

    class BidTransaction {
        -BigDecimal bidAmount
        -LocalDateTime bidTimestamp
    }

    class AntiSnipping {
        +handleAntiSniping(endTime, bidTimestamp)$ LocalDateTime
    }

    class AutoBidConfig {
        -BigDecimal maxBidAmount
        -boolean isActive
        -LocalDateTime createdAt
    }

    BaseEntity <|-- Auction
    BaseEntity <|-- BidTransaction
    BaseEntity <|-- AutoBidConfig

    Auction "1" *-- "1" Item : bids on
    Auction "1" o-- "*" BidTransaction : contains
    Auction "1" --> "0..1" Bidder : winningUser
    Auction "1" *-- "1" AuctionStatus : status
    Auction ..> AntiSnipping : uses
    Auction "1" o-- "*" AutoBidConfig : autoBidConfigs

    BidTransaction "1" --> "1" Item : references
    BidTransaction "1" --> "1" Bidder : made by

    AutoBidConfig "*" --> "1" Auction : auction
    AutoBidConfig "*" --> "1" Bidder : bidder

    %% =======================
    %% NOTIFICATION PACKAGE
    %% =======================
    class Notification {
        -String title
        -String description
        -NotificationType type
        -boolean isRead
        -LocalDateTime createdAt
    }

    BaseEntity <|-- Notification
    Notification "*" --> "1" User : user
    Notification "*" --> "0..1" Auction : auction

    %% =======================
    %% CHAT PACKAGE
    %% =======================
    class ChatMessage {
        -String content
        -boolean isRead
        -LocalDateTime sendTime
    }

    class Connection {
        -String status
        -LocalDateTime createdAt
    }

    BaseEntity <|-- ChatMessage
    BaseEntity <|-- Connection
    ChatMessage "*" --> "1" User : sender
    ChatMessage "*" --> "1" User : receiver
    Connection "*" --> "1" User : sender
    Connection "*" --> "1" User : receiver

    %% =======================
    %% SERVICE PACKAGE
    %% =======================
    class AuctionService {
        <<Singleton>>
        -Map~String, Auction~ auctions
        -static AuctionService instance$
        +getInstance() AuctionService$
        +createAuction(item, seller, start, end) Auction
        +getAuction(id) Auction
        +placeBid(...) boolean
        +getAllAuctions() Map
    }

    class UserService {
        -static Map~String, User~ users$
        +Register(...) boolean
        +Login(...) boolean
        +ForgotPassword(...) boolean
        +update...()
        +setMoney...()
        +banUser(...)
    }

    class PaymentService {
        <<abstract>>
        +addFreezeMoney(...)
        +subFreezeMoney(...)
        +addWalletMoney(...)
        +subWalletMoney(...)
        +FreezeMoney(...)
        +unFreezeMoney(...)
        +transferMoney(...)
    }

    class AdminService {
    }

    class NotificationService {
        +sendNotification(...)
        +markAsRead(...)
    }

    class ChatService {
        +sendMessage(...)
        +createConnection(...)
        +acceptConnection(...)
        +blockUser(...)
    }

    AuctionService "1" o-- "*" Auction : manages
    UserService "1" o-- "*" User : manages
    PaymentService ..> UserService : calls logic
    AuctionService ..> PaymentService : calls logic
    Auction ..> PaymentService : calls logic
    NotificationService ..> Notification : manages
    ChatService ..> ChatMessage : manages
    ChatService ..> Connection : manages

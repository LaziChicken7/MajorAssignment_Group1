    %% BẢNG KẾ THỪA TỪ USER
    users {
        VARCHAR(36) id PK "UUID"
        VARCHAR(255) user_code UK "Mã hiển thị (BID1, SLR2...)"
        VARCHAR(255) user_name UK "Tên đăng nhập"
        VARCHAR(255) email UK
        VARCHAR(255) citizen_id UK "CCCD"
        VARCHAR(255) number_phone
        VARCHAR(255) password
        VARCHAR(255) full_name
        VARCHAR(50) role "ENUM: ADMIN, BIDDER, SELLER"
        VARCHAR(255) avatar_url "URL avatar"
        BOOLEAN is_banned "Bị khóa tài khoản"
    }

    admins {
        VARCHAR(36) id PK,FK "Khóa ngoại trỏ đến users.id"
        VARCHAR(255) department "Mặc định: General"
    }

    bidders {
        VARCHAR(36) id PK,FK "Khóa ngoại trỏ đến users.id"
        VARCHAR(255) address
        VARCHAR(255) bank_account_number
        DECIMAL money_on_wallet "Ví chính"
        DECIMAL moneyin_frozen "Ví đóng băng"
    }

    sellers {
        VARCHAR(36) id PK,FK "Khóa ngoại trỏ đến bidders.id"
        DOUBLE rating
    }

    seller_reviews {
        VARCHAR(36) id PK "UUID"
        VARCHAR(36) seller_id FK "Người bán được đánh giá"
        VARCHAR(36) reviewer_id FK "Người đánh giá"
        INT star "Số sao đánh giá"
        TEXT comment "Bình luận"
        DATETIME created_at "Thời gian đánh giá"
    }

    %% BẢNG KẾ THỪA TỪ ITEM
    items {
        VARCHAR(36) id PK "UUID"
        VARCHAR(255) name
        VARCHAR(255) description
        DATETIME start_time
        DECIMAL start_price
        DECIMAL end_price
        VARCHAR(50) item_type "ENUM: ART, ELECTRONICS, VEHICLE"
        VARCHAR(36) seller_id FK "Người đăng bán"
    }

    item_image_urls {
        VARCHAR(36) item_id FK "Khóa ngoại trỏ đến items.id"
        VARCHAR(255) image_urls "URL ảnh"
    }

    arts {
        VARCHAR(36) id PK,FK "Khóa ngoại trỏ đến items.id"
        VARCHAR(255) name_author
        INT creation_year
    }

    electronics {
        VARCHAR(36) id PK,FK "Khóa ngoại trỏ đến items.id"
        VARCHAR(255) brand
        INT warranty_months
    }

    vehicles {
        VARCHAR(36) id PK,FK "Khóa ngoại trỏ đến items.id"
        VARCHAR(255) engine_type
        INT mileage
    }

    %% BẢNG ĐẤU GIÁ VÀ GIAO DỊCH
    auctions {
        VARCHAR(36) id PK "UUID"
        DATETIME start_time
        DATETIME end_time
        DECIMAL highest_bid "Giá cao nhất hiện tại"
        DECIMAL step_price "Bước giá tối thiểu"
        VARCHAR(50) status "ENUM: OPEN, RUNNING..."
        VARCHAR(50) transaction_status "ENUM: SUCCESS, FAILED, CANCELLED"
        VARCHAR(36) item_id FK,UK "Món hàng (OneToOne)"
        VARCHAR(36) seller_id FK "Người tạo phiên"
        VARCHAR(36) winning_user_id FK "Người dẫn đầu/Thắng cuộc"
    }

    bid_transactions {
        VARCHAR(36) id PK "UUID"
        DECIMAL bid_amount "Mức giá đặt"
        DATETIME bid_timestamp "Thời điểm đặt giá"
        VARCHAR(36) bidder_id FK "Người đặt giá"
        VARCHAR(36) auction_id FK "Thuộc phiên đấu giá nào"
    }

    auto_bid_configs {
        VARCHAR(36) id PK "UUID"
        VARCHAR(36) auction_id FK "Phiên đấu giá"
        VARCHAR(36) bidder_id FK "Người cài đặt"
        DECIMAL max_bid_amount "Mức giá tối đa tự động"
        BOOLEAN is_active "Có đang hoạt động không"
        DATETIME created_at "Thời điểm tạo cấu hình"
    }

    notifications {
        VARCHAR(36) id PK "UUID"
        VARCHAR(36) user_id FK "Người nhận thông báo"
        VARCHAR(36) auction_id FK "Phiên đấu giá liên quan"
        VARCHAR(255) title "Tiêu đề thông báo"
        TEXT description "Nội dung chi tiết"
        VARCHAR(50) type "Phân loại thông báo"
        BOOLEAN is_read "Trạng thái đã đọc"
        DATETIME created_at "Thời điểm nhận"
    }

    connections {
        VARCHAR(36) id PK "UUID"
        VARCHAR(36) sender_id FK "Người gửi yêu cầu"
        VARCHAR(36) receiver_id FK "Người nhận yêu cầu"
        VARCHAR(50) status "Trạng thái kết nối"
        DATETIME created_at "Thời gian tạo"
    }

    chat_messages {
        VARCHAR(36) id PK "UUID"
        VARCHAR(36) sender_id FK "Người gửi tin nhắn"
        VARCHAR(36) receiver_id FK "Người nhận tin nhắn"
        TEXT content "Nội dung cuộc chat"
        BOOLEAN is_read "Trạng thái đọc tin"
        DATETIME send_time "Thời gian gửi"
    }

    %% ==========================================
    %% QUAN HỆ KẾ THỪA (Inheritance)
    %% ==========================================
    users ||--o| admins : "kế thừa (JOINED)"
    users ||--o| bidders : "kế thừa (JOINED)"
    bidders ||--o| sellers : "kế thừa (JOINED)"

    items ||--o| arts : "kế thừa (JOINED)"
    items ||--o| electronics : "kế thừa (JOINED)"
    items ||--o| vehicles : "kế thừa (JOINED)"

    %% ==========================================
    %% QUAN HỆ NGHIỆP VỤ (Business Logic Relationships)
    %% ==========================================
    sellers ||--o{ items : "đăng bán món hàng (1-N)"
    
    items ||--o| auctions : "được đấu giá (1-1)"
    sellers ||--o{ auctions : "tạo ra phiên đấu giá (1-N)"
    bidders ||--o{ auctions : "có thể thắng nhiều phiên (1-N)"
    
    auctions ||--o{ bid_transactions : "chứa các lượt đặt giá (1-N)"
    bidders ||--o{ bid_transactions : "thực hiện lượt đặt giá (1-N)"

    %% Các quan hệ mới tích hợp:
    items ||--o{ item_image_urls : "lưu danh sách ảnh (1-N)"
    sellers ||--o{ seller_reviews : "được đánh giá bởi reviewer (1-N)"
    users ||--o{ seller_reviews : "viết đánh giá (1-N)"
    auctions ||--o{ auto_bid_configs : "cài đặt đấu giá tự động (1-N)"
    bidders ||--o{ auto_bid_configs : "sở hữu cấu hình autobid (1-N)"
    users ||--o{ notifications : "nhận thông báo (1-N)"
    auctions ||--o{ notifications : "liên quan tới phiên (1-N)"
    users ||--o{ chat_messages : "gửi/nhận tin nhắn (1-N)"
    users ||--o{ connections : "thiết lập kết nối chat (1-N)"
